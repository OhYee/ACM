# 题解
求移动最少的数字使序列递增  
首先本身递增的自然就不必再移动了,要想让移动最少就应该先找到 **最长上升子序列**  
套用 dp 模板即可  

然后剩下的数字都是需要移动的,要最少只需要一次将他们放到最终位置.  

也即答案为 `n - dp`

#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
 
const int maxn = 1005;
int node[maxn];
int ans[maxn];
 
int main(){
    cin.tie(0);
    cin.sync_with_stdio(false);
     
    int n;
    while(cin>>n){
        for(int i=1;i<=n;i++)
            cin>>node[i];
         
        int len = 1;
        ans[1] = node[1];
        for(int i = 2; i <= n; ++i) {
         if(node[i] > ans[len])
              ans[++len] = node[i];
            else
              *lower_bound(ans + 1,ans + 1 + len,node[i]) = node[i];
        }
 
        cout<< n - len<<endl;
    }
}
